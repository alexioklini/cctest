spacy>=3.8,<4.0
# German NER model pinned to a specific release wheel. Hosted on GitHub
# releases (stable, no runtime API dependency). The 3.8.0 wheel matches
# the spaCy 3.8 pin above; the `_md` variant ships pre-trained word
# vectors on top of the same tok2vec + ner pipeline as `_sm` (~45 MB on
# disk, ~120 MB resident with parser/tagger/lemmatizer disabled).
# Quality lift over `_sm` is most visible on casual / mixed-case text
# where the smaller model over-fires PER on lowercase function-words.
de_core_news_md @ https://github.com/explosion/spacy-models/releases/download/de_core_news_md-3.8.0/de_core_news_md-3.8.0-py3-none-any.whl
# M9.3 (G12): English NER recall net. The KYC/DD corpus is majority non-German;
# scan_text unions de + en PERSON/GPE/ORG findings (OntoNotes labels for en).
en_core_web_md @ https://github.com/explosion/spacy-models/releases/download/en_core_web_md-3.8.0/en_core_web_md-3.8.0-py3-none-any.whl

# PDF extraction: PyMuPDF (fitz) + the pymupdf4llm markdown wrapper. fitz was
# already an implicit dep (engine/doc_convert._extract_pdf); pymupdf4llm renders
# tables/layout to markdown far better than markitdown on financial reports
# (see the conversion matrix in Settings → Service-Modelle). NOTE: AGPL-3.0
# (Artifex) — clear commercial licensing for the deployment.
pymupdf
pymupdf4llm

# Macro-enabled + binary Excel workbooks: .xlsb needs pyxlsb (openpyxl can't read
# the binary format); oletools/olevba extracts VBA macro SOURCE (never executed)
# from .xlsm/.xls/.xlsb so the agent can read automation logic. .xlsm cells are
# read by openpyxl like any .xlsx.
pyxlsb
oletools

# Diagram rendering (render_diagram tool): mermaid-cli is a NODE dep, not pip —
# see diagram_render/package.json. One-time per machine: `cd diagram_render && npm install`
# (bundles its own Chromium via Puppeteer). reportlab/python-docx/python-pptx (already
# used by write_document) handle embedding the rendered chart into pdf/docx/pptx.

sqlglot>=20.0.0
