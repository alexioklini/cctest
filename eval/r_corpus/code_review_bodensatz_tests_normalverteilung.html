#' @param file_path Character, path to the Excel file.

  tryCatch({

    message("Fehler beim Laden der Daten: ", e$message)

#' Run normality tests on data

#' @param data Numeric vector, data to test.
#' @param test_type Character, type of test to run ("ks", "jarque", "shapiro", "sf").
#' @return List of test results.

  }
  if (test_type == "all" || test_type == "sf") {
    results$sf <- sf.test(data)
  }
  
  return(results)

#' Create visualizations for data
#'
#' @param data Numeric vector, data to visualize.
#' @param group Character, name of the group (e.g., "EUR_Firma").
#' @return List of ggplot objects.
create_visualizations <- function(data, group) {
  visualizations <- list()
  
  # Zeitreihendiagramm
  visualizations$time_series <- ggplot(data = data.frame(value = data), mapping = aes(x = seq_along(data), y = value)) +
    geom_line() +
    labs(title = paste("Bestandsveränderung -", group), x = "Zeit", y = "Bestandshöhe")
  
  # Histogramm
  visualizations$histogram <- hist(data, main = paste("Histogramm -", group))
  
  # QQ-Plot
  visualizations$qqplot <- qqnorm(data, main = paste("QQ Plot -", group))
  qqline(data)
  
  return(visualizations)
