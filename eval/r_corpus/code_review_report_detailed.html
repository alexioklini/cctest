Haupterkenntnisse:

- Kritischste Probleme: Hardcodierte absolute Windows-Pfade in allen sechs Skripten machen sie nicht-portabel und blockieren jede Team-Kollaboration. Fehlende Fehlerbehandlung führt zu unkontrollierten Abstürzen bei invaliden Eingabedaten.

- Performance-Schwächen: TL_122025.R verwendet eine explizite for-Schleife über alle Deals statt vektorisierter Operationen.

Der Report liefert zu jedem Befund konkrete Vorher-/Nachher-Codebeispiele, empfiehlt eine moderne Projektstruktur mit renv/here/testthat und enthält eine priorisierte Umsetzungs-Checkliste.

#' CSV-Datei einlesen und optional jährliche PDs auf monatliche interpolieren
#'
#' @param filename Pfad zur CSV-Datei
#' @param convmat Logical: Ergebnis als Matrix zurückgeben?
#' @param kopf Logical: Hat die CSV einen Header?
#' @param readtime Integer: 1 = Zeitdimension parsen (Jahre → Monate interpolieren)
#' @param removecolcsv Integer: Spaltenindex zum Entfernen (0 = keine)
#' @param pdtime Character: "PD_input_monthly" oder "PD_input_yearly"
#' @param interpol Character: "first_order" oder "second_order"
#' @return Matrix mit (interpolierten) PD-Werten
readcsv2 <- function(filename, convmat = TRUE, kopf = FALSE, readtime = 0,
                     removecolcsv = 0, pdtime = "PD_input_monthly",
                     interpol = "first_order") {

  # Dateiendung prüfen
  file_ext <- tools::file_ext(filename)
  if (file_ext != "csv") {
    stop("readcsv2 unterstützt nur CSV-Dateien. Erhalten: ", file_ext)
  }

  result <- read.table(filename, sep = ";", header = kopf)

  if (removecolcsv > 0) {
    result <- result[, -removecolcsv, drop = FALSE]
  }

  time <- "Months"
  if (readtime == 1 && pdtime == "PD_input_yearly") {
    time <- "Years"
  }

  if (convmat) {
    result <- as.matrix(result)
  }

  # Interpolation von jährlichen auf monatliche Werte
  if (time == "Years") {
    M_cpd_y <- result
    M_cpd_y <- cbind(0, M_cpd_y)  # Vorspalte mit Nullen
    M_cpd <- NULL

    n_year_pairs <- ncol(M_cpd_y) - 1
    for (i in seq_len(n_year_pairs)) {
      for (j in 0:11) {
        if (interpol == "first_order") {
          # Lineare Interpolation zwischen Jahr i und i+1
          interp_val <- M_cpd_y[, i] * (1 - j / 12) + M_cpd_y[, i + 1] * (j / 12)
          M_cpd <- cbind(M_cpd, interp_val)

        } else if (interpol == "second_order") {
          # Lagrange-Interpolation 2. Ordnung über 3 Stützstellen
          jj <- j
          t1 <- i
          if (i == n_year_pairs) {
            jj <- jj + 12
            t1 <- t1 - 1
          }
          L0 <- (jj / 12 - 1) * (jj / 12 - 2) / 2
          L1 <- (jj / 12) * (2 - jj / 12)
          L2 <- (jj / 12) * (jj / 12 - 1) / 2
          t2 <- t1 + 1
          t3 <- t1 + 2
          interp_val <- L0 * M_cpd_y[, t1] + L1 * M_cpd_y[, t2] + L2 * M_cpd_y[, t3]
          M_cpd <- cbind(M_cpd, interp_val)

        } else {
          stop("Unbekannte Interpolationsmethode: ", interpol)
        }
      }
    }
    result <- M_cpd[, -1, drop = FALSE]  # Vorspalte entfernen
  }

  return(result)
}

#' 12-Monats Expected Credit Loss berechnen
#'
#' @param month_remaining Integer: Restlaufzeit in Monaten
#' @param rating Integer: Rating-Index für PD_cube
#' @param exposure Numeric: Exposure at Default
#' @param lgd Numeric: Loss Given Default
#' @param eir Numeric: Effective Interest Rate
#' @param pd_cube Matrix/Array: PD-Kurven
#' @param port Integer: Portfolio-Index (default 1)
#' @return Numeric: 12-Monats-ECL
get_12M_ECL <- function(month_remaining, rating, exposure, lgd, eir, pd_cube, port = 1) {
  stopifnot(
    is.numeric(month_remaining), month_remaining >= 0,
    is.numeric(rating), rating >= 1,
    is.numeric(exposure), exposure >= 0,
    is.numeric(lgd), lgd >= 0, lgd <= 1,
    is.numeric(eir), eir >= 0
  )

  if (month_remaining == 0) return(0)

  years_effective <- min(month_remaining / 12, 1)
  rem_time <- min(month_remaining, 12)

  # PD_cube-Zugriff mit Boundary-Check
  if (rating > nrow(pd_cube) || rem_time > ncol(pd_cube)) {
    stop("Rating ", rating, " oder Laufzeit ", rem_time, " außerhalb PD_cube-Dimensionen")
  }
  pd_value <- pd_cube[rating, rem_time, port]

  exposure * lgd * pd_value / (1 + eir)^years_effective
}

                        exposure = NULL, lgd = NULL, eir = NULL, amort = "None",
                        pd_cube, port = 1, mode = c("flat", "columns")) {

  mode <- match.arg(mode)

  if (mode == "columns") {
    # Modus: EAD/LGD aus Dataframe-Spalten
    month_remaining <- row_i$Month_Remaining
    rating <- row_i$Rating
    eir <- row_i$EIR
  }

  stopifnot(
    is.numeric(month_remaining), month_remaining >= 0,
    is.numeric(rating), rating >= 1,
    is.numeric(eir), eir >= 0
  )

  if (month_remaining == 0) return(0)

  full_years <- floor(month_remaining / 12)
  years_effective <- month_remaining / 12
  d_ecl <- 0.0

  # Volle Jahre
  if (full_years >= 1) {
    for (j in seq_len(full_years)) {
      pd_current <- pd_cube[rating, j * 12, port]
      pd_previous <- if (j > 1) pd_cube[rating, (j - 1) * 12, port] else 0
      marginal_pd <- pd_current - pd_previous

      if (mode == "columns") {
        ead_cols <- grep("EAD_", colnames(row_i))
        lgd_cols <- grep("LGD_", colnames(row_i))
        ead_val <- row_i[1, ead_cols[(j - 1) * 12 + 1]]
        lgd_val <- row_i[1, lgd_cols[(j - 1) * 12 + 1]]
      } else {
        ead_factor <- get_ead_f(eir, years_effective, j - 1, amort)
        ead_val <- exposure * ead_factor
        lgd_val <- lgd
      }

      d_ecl <- d_ecl + ead_val * lgd_val * marginal_pd / (1 + eir)^j
    }
  }

  # Restmonate (unvollständiges letztes Jahr)
  pd_current <- pd_cube[rating, month_remaining, port]
  pd_previous <- if (full_years >= 1) pd_cube[rating, full_years * 12, port] else 0

  if (years_effective > full_years) {
    if (mode == "columns") {
      ead_cols <- grep("EAD_", colnames(row_i))
      lgd_cols <- grep("LGD_", colnames(row_i))
      ead_val <- row_i[1, ead_cols[full_years * 12 + 1]]
      lgd_val <- row_i[1, lgd_cols[full_years * 12 + 1]]
    } else {
      ead_factor <- get_ead_f(eir, years_effective, full_years, amort)
      ead_val <- exposure * ead_factor
      lgd_val <- lgd
    }
  } else {
    ead_val <- 0
    lgd_val <- 0
  }

  marginal_pd <- pd_current - pd_previous
  d_ecl <- d_ecl + ead_val * lgd_val * marginal_pd / (1 + eir)^years_effective

  return(d_ecl)
}

#' Exposure-at-Default-Faktor berechnen
#'
#' @param eir Numeric: Effective Interest Rate
#' @param total_years Numeric: Gesamtlaufzeit in Jahren
#' @param year_index Integer: Aktuelles Jahr (0-basiert)
#' @param amortization Character: "Annuity", "Linear" oder "None"
#' @return Numeric: EAD-Faktor (0 bis 1)
get_ead_f <- function(eir, total_years, year_index, amortization = "None") {
  stopifnot(
    is.numeric(eir), eir >= 0,
    is.numeric(total_years), total_years > 0,
    is.numeric(year_index), year_index >= 0
  )

  amortization <- match.arg(amortization, c("Annuity", "Linear", "None"))

  factor <- switch(amortization,
    Annuity = {
      i_adj <- year_index + 1
      ((1 + eir)^total_years - (1 + eir)^(i_adj - 1)) / ((1 + eir)^total_years - 1)
    },
    Linear = {
      1 - (1 / total_years) * year_index
    },
    None = 1
  )

  max(0, factor)
}

Berechnet kumulative Through-the-Cycle-PD-Kurven aus einer jährlichen Migrationsmatrix (1Y_mm_finapu_2025.csv). Führt Matrixmultiplikation für Multi-Year-Projektion durch und interpoliert auf monatliche Werte. Enthält Rating-Interpolation über logarithmische Skalierung.

ProblemBeschreibungKritikalität
Berechnungsfehler?Zeile 18: lt_cPD[,i](1-j/12)+lt_cPD[,i+1](j/12) – die Interpolation verwendet Spalte i (alt) und i+1 (neu). Aber lt_cPD wurde gerade in Zeile 16 um t erweitert. Aber in Iteration 2 hat lt_cPD 3 Spalten: Jahr-1, Jahr-2, Jahr-3.
